
clc;
clear all;
close all;

% _analog signal
% x(t) = 5 sin ( 2 * pi * 2 * t + phase)

a = 5;
f = 2;
fs = 50 * f;
T = 2;
ph = pi / 3; % 90 degree

t = 0:1/fs:T-(1/fs);

x = a * sin (2 * pi * f * t + ph);

hold on;
subplot(2,1,1);
plot(t, x);
xlabel("T");
ylabel("x(t)");
title("Analog Signal Sine Wave with 60 Degree Phase");


% _digital signal
% x(nt) = 5 sin ( 2 * pi * f/fs * n + phase)
% same as previous, just plot as a stem plot

subplot(2,1,2);
stem(x);
xlabel("nt");
ylabel("x(nt)");
title("Digital Signal Sine Wave with 60 Degree Phase");
